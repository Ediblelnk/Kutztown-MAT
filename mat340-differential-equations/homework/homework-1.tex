\subsection*{1.1}
In Problems 1-12, a differential equation is given along with the field or problem area in which it arises. Classify each as an ordinary differential equation (ODE) or a partial differential equation (PDE), give the order, and indicate the independent and dependent variables. If the equation is an ordinary differential equation, indicate whether the equation is linear or nonlinear.
\begin{enumerate}
    \qitem{1}{$5\frac{d^2 x}{d t^2} + 4\frac{dx}{dt} + 9x = 2\cos 3t$}{ODE linear, order 2, independent vars are $t$, dependent vars are $x$}
    \qitem{2}{$\frac{d^2 y}{d x^2} -2x\frac{dy}{dx} + 2y = 0$}{ODE nonlinear, order 2, independent vars are $x$, dependent vars are $y$}
    \qitem{3}{$\frac{dy}{dx} = \frac{y(2-3x)}{x(1-3y)}$}{ODE nonlinear, order 1, independent vars are $x$, dependent vars are $y$}
    \qitem{4}{$\frac{\partial^2 u}{\partial x^2} + \frac{\partial^2 u}{\partial y^2} = 0$}{PDE, order 2, independent vars are $x$, dependent vars are $y$}
    \qitem{5}{$y \left[1 + \left(\frac{dy}{dx}\right)^2\right] = C$}{ODE nonlinear, order 1, independent vars are $x,y$, dependent vars are $u$}
    \qitem{6}{$\frac{dx}{dt} = k(4-x)(1-x)$, where $k$ is a constant}{ODE nonlinear, order 1, independent vars are $t$, dependent vars are $x$}
    \qitem{7}{$\frac{dp}{dt} = kp(P-p)$, where $k$ and $P$ are constants}{ODE nonlinear, order 1, independent vars are $t$, dependent vars are $p$}
    \qitem{8}{$\sqrt{1-y}\frac{d^2 y}{dx^2} + 2x\frac{dy}{dx} = 0$}{ODE nonlinear, order 2, independent vars are $x$, dependent vars are $y$}
    \qitem{9}{$x\frac{d^2 y}{dx^2} + \frac{dy}{dx} + xy = 0$}{ODE nonlinear, order 2, independent vars are $x$, dependent vars are $y$}
    \qitem{10}{$8\frac{d^4 y}{dx^4} = x(1-x)$}{ODE nonlinear, order 4, independent vars are $x$, dependent vars are $y$}
    \qitem{11}{$\frac{\partial N}{\partial t} = \frac{\partial^2 N}{\partial r^2} + \frac{1}{r} \frac{\partial N}{\partial r} + kN$, where $k$ is a constant}{PDE, order 2, independent vars are $t,r$, dependent vars are $N$}
    \qitem{12}{$\frac{d^2 y}{dx^2} - 0.1(1-y^2)\frac{dy}{dx} + 9y = 0$}{ODE nonlinear, order 2, independent vars are $x$, dependent vars are $y$}
\end{enumerate}

In Problems 3-8, determine whether the given function is a solution to the given differential equation.
\begin{enumerate}
    \qitem{3}{$y = \sin x + x^2$,\qquad$\frac{d^2y}{dx^2} + y = x^2 + 2$}{answer}
    \qitem{4}{$x = 2\cos t - 3\sin t$,\qquad$x'' + x = 0$}{answer}
    \qitem{5}{$\theta = 2e^{3t} - e^{2t}$,\qquad$\frac{d^2\theta}{dt^2} - \theta\frac{d\theta}{dt} + 3\theta = -2e^{2t}$}{answer}

In Problems 9-13, determine whether the given relation is an implicit solution to the given differential equation. Assume that the relationship does define y implicitly as a function of $x$ and use implicit differentiation.
