\section{Real Vector Spaces}
A \bld{real vector space} is simply a \itl{nonempty set} that satisfies 10
properties called \bld{10 axioms of a real vector space}.
\begin{itemize}
  \item $\vec{v} \in $ vector space $V$ can be \itl{anything}
  \item \bld{Never} assume that an element $\vec{v} \in V$ is an ordered pair
\end{itemize}

\subsection*{Addition}
\begin{itemize}
  \item denoted by $\oplus$
  \item simply a map
\end{itemize}
\[
  \oplus: V \times V \rightarrow V
\]
Example of a definition of $\oplus$ for $V=\{\text{apple}, \text{orange}, \text{banana}\}$:
\begin{center}
  \begin{tabular}{c|ccc}
    $\oplus$ & apple  & orange       & banana \\
    \hline
    apple    & banana & \bld{banana} & apple  \\
    orange   & orange & apple        & banana \\
    banana   & banana & orange       & orange
  \end{tabular}
  \[
    \oplus(\text{apple}, \text{orange}) = \text{banana} = \text{apple} \oplus \text{orange}
  \]
\end{center}

\subsection*{Scalar Multiplication}
\begin{itemize}
  \item denoted by $\odot$
  \item simply a map
  \item \itl{must} be $r \times \vec{v}$ for $r \in \bb{R}, \vec{v} \in V$
\end{itemize}
\[
  \odot: \bb{R} \times V \rightarrow V
\]
Example of a definition of $\odot$ for $V=\{\text{apple}, \text{orange}, \text{banana}\}$:
\begin{center}
  \begin{align*}
    k \odot \text{apple}  & = \text{orange}, \forall~ k \in \bb{R} \\
    k \odot \text{orange} & = \left\{
    \begin{array}{lr}
      \text{orange}, & \text{if}~k \leq 2, \\
      \text{banana}, & \text{if}~k > 2,    \\
    \end{array} \right.                          \\
    k \odot \text{banana} & = \left\{
    \begin{array}{ll}
      \text{banana}, & \text{if}~k < -5\sqrt{2},          \\
      \text{apple},  & \text{if}~-5\sqrt{2} \leq k < 1.2, \\
      \text{banana}, & \text{if}~k = 1.2,                 \\
      \text{orange}, & \text{if}~k > 2,                   \\
    \end{array}
    \right.
  \end{align*}
  \[
    \odot(3, \text{orange}) = \text{banana} = 3 \odot \text{orange}
  \]
\end{center}

\subsection*{10 Good Properties of Addition and Scalar Multiplication}
\begin{enumerate}
  \item \bld{Closed Under Addition} $\forall~\vec{v}, \vec{u} \in V$,
        \[
          \vec{u} \oplus \vec{v} \in V
        \]
  \item \bld{Commutativity Under Addition} $\forall~\vec{v}, \vec{u} \in V$,
        \[
          \vec{u} \oplus \vec{v} = \vec{v} \oplus \vec{u}
        \]
  \item \bld{Associativity Under Addition} $\forall~\vec{v}, \vec{u}, \vec{w} \in V$,
        \[
          \vec{u} \oplus (\vec{v} \oplus \vec{w}) = (\vec{u} \oplus \vec{v}) \oplus \vec{w}
        \]
  \item \bld{Additive Identity Exists} $\exists~\vec{u}~\forall~\vec{v} \in V$,
        \[
          \vec{u} \oplus \vec{v} = \vec{v} \oplus \vec{u} = \vec{v} \\
        \]
        \[
          \vec{u}~\text{is called the additive identity}, \id
        \]
  \item \bld{Additive Inverse Always Exists} $\forall~\vec{v}~\exists~\vec{w} \in V$,
        \[
          \vec{v} \oplus \vec{w} = \vec{w} \oplus \vec{v} = \id
        \]
        \[
          \vec{w} = -\vec{v}~\text{and is pronounced as \itl{bar-$\vec{v}$}}
        \]
  \item \bld{Closed Under Scalar Multiplication} $\forall~k \in \bb{R}, \vec{v} \in V$,
        \[
          k \odot \vec{v} \in V
        \]
  \item \bld{Distributivity Over $\oplus$} $\forall~k \in \bb{R}, \vec{u}, \vec{v} \in V$,
        \[
          k \odot (\vec{u} \oplus \vec{v}) = k \odot \vec{u} \oplus k \odot \vec{v}
        \]
  \item \bld{Distributivity Over $+$} $\forall~k, \ell \in \bb{R}, \vec{v} \in V$,
        \[
          (k + \ell) \odot \vec{v} = k \odot \vec{v} \oplus \ell \odot \vec{v}
        \]
  \item \bld{Associativity Over Scalar Multiplication} $\forall~k, \ell \in \bb{R}, \vec{v} \in V$,
        \[
          (k \cdot \ell) \odot \vec{v} = k \odot (\ell \odot \vec{v})
        \]
  \item \bld{1 Fixes Every Element In V By $\odot$} $\forall~\vec{v} \in V$,
        \[
          1 \odot \vec{v} = \vec{v}
        \]
\end{enumerate}
\subsection*{Tips To Remember The 10 Axioms}
\begin{itemize}
  \item first 5 axioms deal with addition ONLY, the next 5 axioms involve scalar multiplication
  \item first of the 5 axioms for addition and scalar multiplication deal with closure
  \item axioms 4 and 5 are about the existence of something
  \item axioms 8 and 9 are the only axioms that involve 2 real numbers
\end{itemize}
\subsection*{Verifying the 10 Axioms}
\begin{itemize}
  \bitem{a.} Axioms (1) and (6): proof of closure
  \bitem{b.} Axioms (4) and (5): show existence
  \bitem{c.} Axioms (2), (3), (7), (8), (9), (10): proof for all elements
\end{itemize}
\subsection*{Standard Addition and Scalar Multiplication for $\bb{R}^n$}
$\forall~\vec{u} = (u_1, u_2, \ldots, u_n), \vec{v} = (v_1, v_2, \ldots, v_n) \in V$ and $\forall~k \in \bb{R}$,
\begin{align*}
  \vec{u} \oplus \vec{v} & = (u_1 + v_1, u_2 + v_2, \ldots, u_n + v_n) \\
  k \odot \vec{u}        & = (ku_1, ku_2, \ldots, ku_n)
\end{align*}
