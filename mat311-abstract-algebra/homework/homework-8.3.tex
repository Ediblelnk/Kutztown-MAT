In Exercises 30 through 34, determine whether the given function is a permutation of $\bb{R}$.
\begin{enumerate}
    \qitem{30}{$f_1: \bb{R} \rightarrow \bb{R}$ defined by $f_1(x) = x+1$}{$f_1(x) = x+1$ \bld{is} a permutation.}
    \qitem{31}{$f_2: \bb{R} \rightarrow \bb{R}$ defined by $f_2(x) = x^2$}{$f_2(x) = x^2$ \bld{is not} a permutation because it is not one-to-one or onto.}
    \qitem{32}{$f_3: \bb{R} \rightarrow \bb{R}$ defined by $f_3(x) = -x^3$}{$f_3(x) = -x^3$ \bld{is} a permutation.}
    \qitem{33}{$f_4: \bb{R} \rightarrow \bb{R}$ defined by $f_4(x) = e^x$}{$f_4(x) = e^x$ \bld{is not} a permutation because it is not onto.}
    \qitem{35}{Mark each of the following:}{
        True or False
        \begin{enumerate}
            \qitem{a}{Every permutation is a ono-to-one function}{True, but it is also onto.}
            \qitem{b}{Every function is a permutation if and only if it is one-to-one}{False, it must also be onto.}
            \qitem{c}{Every function from a finite set onto itself must be one to one}{True}
            \qitem{d}{Every group $G$ is isormorphic to a subgroup of $S_G$}{True, this is Cayley's Theorem}
        \end{enumerate}
    }
\end{enumerate}
\bld{Theory}

In Exercises 40 through 43, let $A$ be a set, $B$ a subset of $A$, and let $b$ be one particular element of $B$. Determine whether the given set is sure to be a subgroup of $S_A$ under the induced operation. Here $\sigma[B] = \{\sigma(x) : x \in B\}$.
\begin{enumerate}
    \qitem{40}{$\{\sigma \in S_A : \sigma(b) = b\}$}{Yes}
    \qitem{41}{$\{\sigma \in S_A : \sigma(b) \in B\}$}{No}
    \qitem{42}{$\{\sigma \in S_A : \sigma[B] \subseteq B\}$}{No}
\end{enumerate}
