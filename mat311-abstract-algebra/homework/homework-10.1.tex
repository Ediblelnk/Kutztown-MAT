    \qitem{6}{Find all the left cosets of the subgroup $\{\rho_0,\mu_2\}$ of the group $D_4$ given by Table 8.12.
    \begin{center}
        \includegraphics*[width=200px]{assets/table8.12.png}
    \end{center}

        \begin{align*}
            \rho_0H & = \{\rho_0, \mu_2\} = \mu_2H \\
            \rho_1H & = \{\rho_1, \delta_2\} = \delta_2H \\
            \rho_2H & = \{\rho_2, \mu_1\} = \mu_1H \\
            \rho_3H & = \{\rho_3, \delta_1\} = \delta_1H

            H\rho_0 & = \{\rho_0,\rho_2\} = H\rho_2 \\
            H\rho_1 & = \{\rho_1,\rho_3\} = H\rho_3 \\
            H\mu_1 & = \{\mu_1,\mu_2\} = H\mu_2 \\

    \qitem{13}{Find the index of $struct{\mu_1}$ in the group $S_3$, using notation of Example 10.7}{

    \qitem{15}{Let $\sigma = (1,2,5,4)(2,3)$ in $S_5$. Find the index of $\struct{\sigma}$ in $S_5$.}{
        According to the Theorem of Lagrange, $|G:H| = \frac{|G|}{|H|}$. In this instance, $G = S_5$, so $|G| = 5! = 120$. $\sigma = (1,2,5,4)(2,3) = (1,2,3,5,4)$, so $\sigma^5 = \sigma^0$. This means $|\struct{\sigma}| = 5$. Thus we have the following.
        \[
            |G:H| = \frac{|G|}{|H|} = \frac{120}{5} = 24
        \]
    }
